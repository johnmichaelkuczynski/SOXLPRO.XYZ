│   ├── cc-engine.ts                         # CROSS-CHUNK COHERENCE ENGINE (~720 lines)
│   │                                        # Three-pass architecture for F10 batch / long inputs:
│   │                                        #
│   │                                        # PASS 1 - SKELETON EXTRACTION
│   │                                        #   extractSkeleton(input, model) → FormalSkeleton
│   │                                        #   { primitives[], axioms[], definitions[], constants[] }
│   │                                        #
│   │                                        # PASS 2 - CHUNK PROCESSING
│   │                                        #   processChunk(domain, skeleton, dbInstructions) → ChunkDelta
│   │                                        #   - Each F10 domain processed independently
│   │                                        #   - Uses domain-specific DB preset (f10-mathematical, f10-chemical, etc.)
│   │                                        #   - MissingPresetError thrown if preset missing (fail-fast, no silent fallback)
│   │                                        #
│   │                                        # PASS 3 - STITCH VERIFICATION
│   │                                        #   verifyCompleteness(skeleton, deltas) → CCViolation[]
│   │                                        #   - Ensures every axiom + primitive was addressed
│   │                                        #   - Reports missing-axiom, missing-primitive, contradiction violations
│   │
│   │                                        # KEY HELPERS:
│   │                                        # - normalizeFunctionType(s) → canonical hyphen-separated DB key
│   │                                        #   (handles "F10", "Function 10:", "Find an Interpretation", etc.)
│   │                                        # - getDBInstructions(functionType, subfunction, required)
│   │                                        #   throws MissingPresetError when required and not found
│   │                                        # - getDBInstructionsForDomain(domain) - F10 domain-specific (NO fallback)
│   │                                        # - needsCC(input, functionType) - triggers CC when axiom count
│   │                                        #   or input length crosses CC_THRESHOLDS
│   │                                        # - parseInterpretation(output, domain) - structured extraction
│   │                                        #   for mt_interpretations rows
│   │
│   │                                        # CONSTANTS:
│   │                                        # - F10_DOMAINS[] (23 domains, grouped A–E)
│   │                                        # - DOMAIN_TO_DB_KEY{} (CC domain name → f10-* preset key)
│   │                                        # - SHORT_FORM_MAP{0..10} (integer function id → DB key)
│   │                                        # - CC_THRESHOLDS { minAxiomCount, minInputLength }
│   │
│   │                                        # PUBLIC ENTRYPOINTS:
│   │                                        # - runWithCC(CCRequest) → CCResult
│   │                                        # - runF10BatchMode(input, model, selectedDomains?) → jobId
│   │                                        # - getJobStatus(jobId), getJobChunks(jobId)
